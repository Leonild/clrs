\begin{align}
  \sum_{i=1}^2\sum_{j=1}^nr_i(j) &= 2^{n+1}-2 \\
  \sum_{j=1}^n2^{n-j} &= 2^{n}-1 & \text{by (15.8)} \\
  \sum_{j=1}^n2^{n-j} &= \sum_{j=0}^{n-1}j \\
  &= \frac{2^n-1}{2-1} & \text{geometric series}\\
  &= 2^n-1
\end{align}
